66  	public void loadService() throws Exception {
67  		repository = new SimpleServiceRepository();
68  		provider = new FieldBackedPropertyProvider();
69  		locator = new CumulativeFieldInformerLocator();
70  		locator.add(new BasicFieldInformerLocator());
71  		locator.add(new ServiceBackedFieldLocator(repository));
72  		locator.add(new LazyInterfaceInformerLocator());
73  		reflectiveFactory = new ReflectionBackedInformerFactory(
74  				locator, provider);
75  		proxyInformerFactory = new ProxyBackedInformerFactory(
76  				reflectiveFactory);
77  		
78  		graph = createGraph(graphProvider);
79  		
80  		// Now add some services
81  		repository.add(createServiceFor(Tag.class, TagInformer.class));
82  		repository.add(createServiceFor(Post.class, PostInformer.class));
83  		repository.add(createServiceFor(PostSubClass.class, PostSubClassInformer.class));
84  		repository.add(createServiceFor(User.class, UserInformer.class));
85  		repository.add(createServiceFor(Theme.class, ThemeInformer.class));
86  		
87  		// and reference them
88  		tagService = repository.get(Tag.class);
89  		postService = repository.get(Post.class);
90  		postSubService= repository.get(PostSubClass.class);
91  		userService = repository.get(User.class);
92  		themeService = repository.get(Theme.class);
93  	}
