13   * Global informer factories allow lookup of specific class informers, as long as they're loaded.

15   * @author Nicolas

18  public class ReflectionBackedInformerFactory {
19  	/**
20  	 * Map linking object classes to reflection backed informers used to gather
21  	 * field informations for these classes
22  	 */
23  	private Map<Class<?>, ReflectionBackedInformer<?>> loaded = new HashMap<Class<?>, ReflectionBackedInformer<?>>();
24

26  	 * Pluggable field informer locator
27  	 */
28  	private FieldInformerLocator fieldLocator;
29  	
30  	private PropertyProvider propertyProvider;

32  	public ReflectionBackedInformerFactory(FieldInformerLocator fieldLocator, PropertyProvider provider) {
33  		super();
34  		this.fieldLocator = fieldLocator;
35  		this.propertyProvider = provider;
36  	}
37  
38  	/**
39  	 * Get informer associated to given class
40  	 * 
41  	 * @param <T>
42  	 *            class type
43  	 * @param clazz
44  	 *            input class
45  	 * @return loaded informer for that class
46  	 */
47  	public <T> Informer<T> get(Class<T> clazz) {
48  		if (!loaded.containsKey(clazz)) {
49  			synchronized (this) {
50  				if (!loaded.containsKey(clazz)) {
51  					loaded.put(clazz, new ReflectionBackedInformer<T>(clazz,
52  							this, propertyProvider));
53  				}
54  			}
55  		}
56  		return (Informer<T>) loaded.get(clazz);
57  	}
58  
59  	/**
60  	 * This method generates {@link FieldInformer} for each known field type of
61  	 * object. Furthermore, generated field informer is cloned to make sure it's possible to alter its path
62  	 * 
63  	 * @param field
64  	 *            input field
65  	 * @return a {@link FieldInformer} if possible, an
66  	 *         {@link UnsupportedOperationException} is fired elsewhere (this
67  	 *         way, we will think about handling it later ;-)
68  	 */
69  	public FieldInformer getInformerFor(Property field) {
70  		FieldInformer returned = fieldLocator.getInformerFor(field);
71  		return returned;
72  	}
